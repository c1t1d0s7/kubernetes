## 3. 환경변수를 이용한 사용자화

```
docker run --env NODE_PORT=8080 <image>
```

### 1) 환경변수 테스트를 위한 이미지 빌드 및 생성
> index.js
```javascript
const http = require('http');
const os = require('os');
const ip = require('ip');
const dns = require("dns");

console.log(Date());
console.log("...Start My Node.js Application...");

var handler = function(request, response) {
    console.log(Date());
    console.log("Received Request From " + request.connection.remoteAddress);
    response.writeHead(200);
    response.write("Message: " + process.argv[2] + "\n");
    response.write("Hostname: " + os.hostname() + "\n");
    response.write("Platform: " + os.platform() + "\n");
    response.write("Uptime: " + os.uptime() + "\n");
    response.write("IP: " + ip.address() + "\n");
    response.write("DNS: " + dns.getServers() + "\n");
    response.end();
};

var www = http.createServer(handler);
www.listen(process.env.NODE_PORT);
```

> Dockerfile
```dockerfile
FROM node:slim
WORKDIR /usr/src/app
COPY index.js .
RUN npm install ip
ENTRYPOINT ["node", "index.js"]
CMD ["Hello World!"]
EXPOSE 8080-8090/tcp
```

```
docker build -t <ACCOUNT>/myweb:customport .
```

```
docker login
```

```
docker push <ACCOUNT>/myweb:customport
```

### 2) 쿠버네티스 정의에서 환경변수 설정
> mynapp-pod-env.yml
```yaml
apiVersion: v1
kind: Pod
metadata:
  name: mynapp-pod-env
spec:
  containers:
  - image: c1t1d0s7/myweb:customport
    name: mynapp
    env:
    - name: NODE_PORT
      value: "8088"
    ports:
    - containerPort: 8088
      protocol: TCP
```

```
kubectl create -f mynapp-pod-env.yml
```

```
kubectl describe pod mynapp-pod-env
```

```
kubectl port-forward mynapp-pod-env 8088:8088
```

```
curl http://localhost:8088
```
